    structure GroupoidMultitensor.FiniteGroupoidData :
    Type (max (u + 1) (v + 1))

    A finite groupoid presented combinatorially as objects Obj, morphisms Mor, source/target maps, composition (defined when tgt g = src h), identity, and inverse, all subject to the expected groupoid relations.

    • Obj : Type u
    • Mor : Type v
    • finObj : Fintype self.Obj
    • finMor : Fintype self.Mor
    • decEqObj : DecidableEq self.Obj
    • decEqMor : DecidableEq self.Mor
    • src : self.Mor → self.Obj
    • tgt : self.Mor → self.Obj
    • comp (g h : self.Mor) : self.tgt g = self.src h → self.Mor
    • id : self.Obj → self.Mor
    • inv : self.Mor → self.Mor
    • src_id (x : self.Obj) : self.src (self.id x) = x
    • tgt_id (x : self.Obj) : self.tgt (self.id x) = x
    • src_comp (g h : self.Mor) (p : self.tgt g = self.src h) : self.src (self.comp g h p) = self.src g
    • tgt_comp (g h : self.Mor) (p : self.tgt g = self.src h) : self.tgt (self.comp g h p) = self.tgt h
    • src_inv (g : self.Mor) : self.src (self.inv g) = self.tgt g
    • tgt_inv (g : self.Mor) : self.tgt (self.inv g) = self.src g
