    structure BlockSymplecticData (V : Type u_1) (W : Type u_2) [AddCommGroup V] [Module ℝ V] [AddCommGroup W] [Module ℝ W] :
    Type (max u_1 u_2)

    Algebraic data packaging a 2×2 block-matrix symplectic form on $V \oplus W$: an evaluation pairing $\mathrm{eval} : W \otimes V \to \mathbb{R}$, an invertible block endomorphism $B$ on $V$ together with its transpose-inverse $B^{-T}$ on $W$, and a skew form $C$ on $W$ realized via $\tilde C : W \to V$.

    • eval : W →ₗ[ℝ] V →ₗ[ℝ] ℝ
    • B_endo : V →ₗ[ℝ] V
    • B_inv : V →ₗ[ℝ] V
    • hBinv_left (v : V) : self.B_inv (self.B_endo v) = v
    • hBinv_right (v : V) : self.B_endo (self.B_inv v) = v
    • B_inv_t : W →ₗ[ℝ] W
    • hBinvt (w : W) (v : V) : (self.eval (self.B_inv_t w)) v = (self.eval w) (self.B_inv v)
    • C_form : W →ₗ[ℝ] W →ₗ[ℝ] ℝ
    • hC_skew (w₁ w₂ : W) : (self.C_form w₁) w₂ = -(self.C_form w₂) w₁
    • C_tilde : W →ₗ[ℝ] V
    • hCtilde (w₁ w₂ : W) : (self.eval w₁) (self.C_tilde w₂) = (self.C_form w₁) w₂
