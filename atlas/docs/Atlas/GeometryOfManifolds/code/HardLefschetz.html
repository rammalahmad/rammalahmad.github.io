    class HasKahlerHodgeNumbers {Ω : ℕ → Type u_1} {VF : Type u_2} [inst : DifferentialFormSpace Ω VF] extends HasHodgeNumbers :
    Type

    A Kähler manifold with Hodge number data, additionally tracking the integral $\int_M \omega^n$ of the top power of the Kähler form.

    • complexDim : ℕ
    • hodge : ℕ → ℕ → ℕ
    • conjugation_symmetry (p q : ℕ) : hodge Ω VF p q = hodge Ω VF q p
    • star_symmetry (p q : ℕ) : hodge Ω VF p q = hodge Ω VF (complexDim Ω VF - q) (complexDim Ω VF - p)
    • betti : ℕ → ℕ
    • betti_hodge_link (k : ℕ) : betti Ω VF k = ∑ p ∈ Finset.range (k + 1), hodge Ω VF p (k - p)
    • omega_n_integral : ℝ

        class HasLefschetzDecomposition {Ω : ℕ → Type u_1} {VF : Type u_2} [inst : DifferentialFormSpace Ω VF] extends HasCohomologyWithLefschetz :
