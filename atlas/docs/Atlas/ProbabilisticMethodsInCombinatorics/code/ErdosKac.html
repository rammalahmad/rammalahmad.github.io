              theorem ErdosKac.erdos_kac (t : ℝ) :

              Theorem 4.5.3 (Erdős–Kac 1940). For every $t \in \mathbb{R}$, $$\frac{\#\{x \le n : (\nu(x) - \log\log n)/\sqrt{\log\log n} \ge t\}}{n} \to \Pr(Z \ge t)$$ as $n \to \infty$, where $Z \sim \mathcal{N}(0,1)$.
